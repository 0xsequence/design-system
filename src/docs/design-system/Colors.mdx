import { Meta, ColorPalette, ColorItem } from '@storybook/addon-docs/blocks'
import { colorNames } from '../../tokens/color'
import { capitalize, kebabize } from '../../utils/string'
import { getCssVar } from '../../utils/css'

<Meta title="Design System/Colors" />

### Semantic Colors

Semantic colors point to color palettes and are used to give a name to a color that is used in a specific context. For example, `primary` is a semantic color that points to a color in the `brand` palette. This way, if the brand color changes, the `primary` color will change as well.

<ColorPalette>
  {colorNames.map(key => {
    const name = `--seq-color-${kebabize(key)}`
    const lightValue = getCssVar(name, document.getElementById('light-mode'))
    const darkValue = getCssVar(name, document.getElementById('dark-mode'))

    return (
      <ColorItem
        key={key}
        title={key}
        subtitle={name}
        colors={{ 'light': lightValue, 'dark': darkValue }}
      />
    )

})}

</ColorPalette>
